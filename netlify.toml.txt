# Configuración de Netlify para Hospital Susana López de Valencia
[build]
  # Directorio base del proyecto
  base = "."
  
  # Comando de build (si necesitas procesar archivos)
  command = "echo 'Build completado'"
  
  # Directorio de salida
  publish = "."

# Variables de entorno para desarrollo (NO incluir credenciales reales aquí)
[build.environment]
  NODE_ENV = "production"

# Configuración de headers para CORS y seguridad
[[headers]]
  for = "/*"
  [headers.values]
    # Permitir CORS desde dominios autorizados
    Access-Control-Allow-Origin = "*"
    Access-Control-Allow-Methods = "GET, POST, PUT, DELETE, OPTIONS"
    Access-Control-Allow-Headers = "Content-Type, Authorization"
    
    # Headers de seguridad
    X-Frame-Options = "DENY"
    X-Content-Type-Options = "nosniff"
    X-XSS-Protection = "1; mode=block"
    Referrer-Policy = "strict-origin-when-cross-origin"
    
    # CSP para permitir conexiones a Airtable
    Content-Security-Policy = "default-src 'self'; script-src 'self' 'unsafe-inline' 'unsafe-eval'; style-src 'self' 'unsafe-inline'; connect-src 'self' https://api.airtable.com; img-src 'self' data:; font-src 'self';"

# Configuración de redirects
[[redirects]]
  from = "/api/*"
  to = "/.netlify/functions/:splat"
  status = 200

# Configuración para SPA (Single Page Application)
[[redirects]]
  from = "/*"
  to = "/index.html"
  status = 200
  conditions = {Role = ["admin"], Country = ["US"]}

# Configuración específica para archivos HTML
[[headers]]
  for = "*.html"
  [headers.values]
    Cache-Control = "public, max-age=3600"

# Configuración para archivos JavaScript
[[headers]]
  for = "*.js"
  [headers.values]
    Cache-Control = "public, max-age=86400"

# Configuración para archivos CSS
[[headers]]
  for = "*.css"
  [headers.values]
    Cache-Control = "public, max-age=86400"
